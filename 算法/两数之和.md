## 两数之和
var twoSum = function(numbers, target) {
  let left = 0;
  let right = numbers.length - 1;
  let sum;
  while(left <= right) {
    sum = numbers[left] + numbers[right];
    if (sum < target) {
      left++;
    } else if (sum > target) {
      right--;
    } else if (sum === target) {
      return [left + 1, right + 1];
    }
  }
};
